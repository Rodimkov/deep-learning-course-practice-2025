#   batches ():
def batches(x_data, y_data, batch_size):
    idxs = np.arange(len(x_data))
    np.random.shuffle(idxs)
    for start in range(0, len(x_data), batch_size):
        end = start + batch_size
        batch_idxs = idxs[start:end]
        yield x_data[batch_idxs], y_data[batch_idxs]

def relu(x):
    return np.maximum(0, x)
def relu_dx(x):
    return (x > 0).astype(float)

#  Softmax
def softmax(x):
    exps = np.exp(x - np.max(x, axis=1, keepdims=True))
    return exps / exps.sum(axis=1, keepdims=True)

# Cross_entropy
def cross_entropy(y_true, y_pred):
    """
    () = -1/N _{i=1}^{N} _{k=1}^{K} y_{i,k} log(p_{i,k})
    y_true, y_pred: shape (N, K)
    """
    eps = 1e-8
    y_pred = np.clip(y_pred, eps, 1 - eps)
    N = y_true.shape[0]
    return -np.sum(y_true * np.log(y_pred)) / N

losses = []
for epoch in range(epochs):
    #     batch':
    for x_batch, y_batch in batches(x_train, y_train, batch_size):  # batches    
        # Forward
        z1 = np.dot(x_batch, W1) + b1
        z2 = np.dot(z1, W2) + b2
        a2 = relu(z2)
        z3 = np.dot(a2, W3) + b3
        out = softmax(z3)

        # Loss
        losses.append(cross_entropy(y_batch, out))
        # d/dz softmax + cross-entropy
        dz3 = (out - y_batch) / len(x_batch)

        # Backward
        dW3 = np.dot(a2.T, dz3)
        db3 = np.sum(dz3, axis=0, keepdims=True)
        da2 = np.dot(dz3, W3.T)
        dz2 = da2 * relu_dx(z2)

        dW2 = np.dot(z1.T, dz2)
        db2 = np.sum(dz2, axis=0, keepdims=True)
        dz1 = np.dot(dz2, W2.T) 

        dW1 = np.dot(x_batch.T, dz1)
        db1 = np.sum(dz1, axis=0, keepdims=True)

        #  
        W3 -= learning_rate * dW3
        b3 -= learning_rate * db3
        W2 -= learning_rate * dW2
        b2 -= learning_rate * db2
        W1 -= learning_rate * dW1
        b1 -= learning_rate * db1

    print(f'Epoch {epoch+1}/{epochs}, loss={losses[-1]:.8f}')
