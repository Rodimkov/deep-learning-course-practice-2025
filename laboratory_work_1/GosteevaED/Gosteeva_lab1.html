import numpy as np
import time

class NeuralNetwork:
    def __init__(self, input_size, hidden_size, output_size, seed=42):
        """
          :
          input_size   784 (28x28)
          hidden_size  , 300
          output_size   10 ( 0..9)
        """
        np.random.seed(seed)
        #   (He-   ReLU)
        self.W1 = np.random.randn(input_size, hidden_size) * np.sqrt(2.0 / input_size)
        self.b1 = np.zeros((1, hidden_size))
        self.W2 = np.random.randn(hidden_size, output_size) * np.sqrt(2.0 / hidden_size)
        self.b2 = np.zeros((1, output_size))

    # -------------------
    #    loss
    # -------------------
    def relu(self, x):
        return np.maximum(0, x)

    def relu_derivative(self, x):
        return (x > 0).astype(float)

    def softmax(self, x):
        """
        Softmax  :  shape (m, 10) ->  (m, 10)
           .
        """
        exp_x = np.exp(x - np.max(x, axis=1, keepdims=True))
        return exp_x / np.sum(exp_x, axis=1, keepdims=True)

    def cross_entropy(self, y_pred, y_true):
        """
        -  one-hot y_true.
        y_pred: (m, 10), y_true: (m, 10)
            .
        """
        eps = 1e-12
        m = y_true.shape[0]
        clipped = np.clip(y_pred, eps, 1.0 - eps)
        loss = -np.sum(y_true * np.log(clipped)) / m
        return loss

    # -------------------
    #    Forward pass
    # -------------------
    def forward(self, X):
        """
         .     backward.
        X: (m, 784)
         y_pred (m, 10)
        """
        self.X = X                       #  (m, input_size)
        self.z1 = X.dot(self.W1) + self.b1   #     (m, s)  ( q)
        self.a1 = self.relu(self.z1)         #    (m, s)      ( z)
        self.z2 = self.a1.dot(self.W2) + self.b2  #     (m, 10) (v)
        self.a2 = self.softmax(self.z2)          #    (m, 10)         (u)
        return self.a2

    # -------------------
    #   Backpropagation
    # -------------------
    def backward(self, y_true, learning_rate=0.1):
        """
             (SGD).
        y_true: (m, 10)  one-hot
        """
        m = y_true.shape[0]

        # 1)   z2 (v):  softmax + cross-entropy   (a2 - y) / m
        dz2 = (self.a2 - y_true) / m                        # (m, 10)

        # 2)   W2, b2
        dW2 = self.a1.T.dot(dz2)                            # (s, 10)
        db2 = np.sum(dz2, axis=0, keepdims=True)            # (1, 10)

        # 3)   a1 (,  ""   )
        da1 = dz2.dot(self.W2.T)                            # (m, s)

        # 4)  ReLU: d z1 = da1 * ReLU'(z1)
        dz1 = da1 * self.relu_derivative(self.z1)           # (m, s)

        # 5)   W1, b1
        dW1 = self.X.T.dot(dz1)                             # (input_size, s)
        db1 = np.sum(dz1, axis=0, keepdims=True)            # (1, s)

        # 6)   (SGD )
        self.W1 -= learning_rate * dW1
        self.b1 -= learning_rate * db1
        self.W2 -= learning_rate * dW2
        self.b2 -= learning_rate * db2

    # -------------------
    #    / 
    # -------------------
    def predict(self, X):
        """   ( 0..9)"""
        probs = self.forward(X)
        return np.argmax(probs, axis=1)

    def accuracy(self, X, y_true_onehot):
        """:  argmax-"""
        y_pred = self.forward(X)
        return np.mean(np.argmax(y_pred, axis=1) == np.argmax(y_true_onehot, axis=1))

    # -------------------
    #     Train loop
    # -------------------
    def train(self, X_train, y_train_oh, X_test=None, y_test_oh=None,
              epochs=20, batch_size=32, learning_rate=0.1, verbose=True):
        n = X_train.shape[0]
        history = {"loss": [], "train_acc": [], "test_acc": []}

        for epoch in range(1, epochs + 1):
            t0 = time.time()
            # 
            idx = np.random.permutation(n)
            Xs = X_train[idx]
            ys = y_train_oh[idx]

            epoch_loss = 0.0
            for i in range(0, n, batch_size):
                Xb = Xs[i:i+batch_size]
                yb = ys[i:i+batch_size]

                y_pred = self.forward(Xb)
                batch_loss = self.cross_entropy(y_pred, yb)
                epoch_loss += batch_loss * Xb.shape[0]  #   

                self.backward(yb, learning_rate=learning_rate)

            #    
            epoch_loss /= n
            train_acc = self.accuracy(X_train, y_train_oh)
            history["loss"].append(epoch_loss)
            history["train_acc"].append(train_acc)

            if X_test is not None and y_test_oh is not None:
                test_acc = self.accuracy(X_test, y_test_oh)
                history["test_acc"].append(test_acc)
            else:
                test_acc = None

            t1 = time.time()
            if verbose:
                print(f" {epoch}/{epochs} | loss: {epoch_loss:.4f} | "
                      f"train_acc: {train_acc*100:.2f}% | "
                      f"test_acc: {None if test_acc is None else f'{test_acc*100:.2f}%'} | "
                      f"time: {t1-t0:.2f}s")

        return history

def to_one_hot(y, num_classes=10):
    y_onehot = np.zeros((y.shape[0], num_classes))
    y_onehot[np.arange(y.shape[0]), y] = 1
    return y_onehot
