def show_images(dataset, title, num_images=12):
    indices = np.random.choice(len(dataset), num_images, replace=False)
    
    plt.figure(figsize=(12, 8))
    plt.suptitle(title, fontsize=16)
    
    for i, idx in enumerate(indices):
        plt.subplot(3, 4, i + 1)  #   3x4
        image, label = dataset[idx]
        
        #   
        img = image * torch.tensor(std).reshape(3, 1, 1) + torch.tensor(mean).reshape(3, 1, 1)
        img = img.numpy().transpose(1, 2, 0)    #  (C, H, W)  (H, W, C)
        img = np.clip(img, 0, 1)
        
        plt.imshow(img)
        plt.title(classes[label], fontsize=10)
        plt.axis('off')
    
    plt.tight_layout()
    plt.show()

class CNN(nn.Module):
    def __init__(self, num_classes=10):
        super(CNN, self).__init__()

        #   
        self.conv1 = nn.Sequential(
            nn.Conv2d(3, 64, kernel_size=3, padding=1),     #  (3, 32, 32),  (64, 32, 32)
            nn.BatchNorm2d(64),                             #  (64, 32, 32),  (64, 32, 32)
            nn.ReLU(inplace=True),                          #  (64, 32, 32),  (64, 32, 32)
            nn.Conv2d(64, 64, kernel_size=3, padding=1),    #  (64, 32, 32),  (64, 32, 32)
            nn.BatchNorm2d(64),                             #  (64, 32, 32),  (64, 32, 32)
            nn.ReLU(inplace=True),                          #  (64, 32, 32),  (64, 32, 32)
            nn.MaxPool2d(kernel_size=2, stride=2),          #  (64, 32, 32),  (64, 16, 16)
            nn.Dropout(0.25)                                #  (64, 16, 16),  (64, 16, 16)
        )

        #   
        self.conv2 = nn.Sequential(
            nn.Conv2d(64, 128, kernel_size=3, padding=1),   #  (64, 16, 16),  (128, 16, 16)
            nn.BatchNorm2d(128),                            #  (128, 16, 16),  (128, 16, 16)
            nn.ReLU(inplace=True),                          #  (128, 16, 16),  (128, 16, 16)
            nn.Conv2d(128, 128, kernel_size=3, padding=1),  #  (128, 16, 16),  (128, 16, 16)
            nn.BatchNorm2d(128),                            #  (128, 16, 16),  (128, 16, 16)
            nn.ReLU(inplace=True),                          #  (128, 16, 16),  (128, 16, 16)
            nn.MaxPool2d(kernel_size=2, stride=2),          #  (128, 16, 16),  (128, 8, 8)
            nn.Dropout(0.25)                                #  (128, 8, 8),  (128, 8, 8)
        )

        #   
        self.conv3 = nn.Sequential(
            nn.Conv2d(128, 256, kernel_size=3, padding=1),  #  (128, 8, 8),  (256, 8, 8)
            nn.BatchNorm2d(256),                            #  (256, 8, 8),  (256, 8, 8)
            nn.ReLU(inplace=True),                          #  (256, 8, 8),  (256, 8, 8)
            nn.Conv2d(256, 256, kernel_size=3, padding=1),  #  (256, 8, 8),  (256, 8, 8)
            nn.BatchNorm2d(256),                            #  (256, 8, 8),  (256, 8, 8)
            nn.ReLU(inplace=True),                          #  (256, 8, 8),  (256, 8, 8)
            nn.MaxPool2d(kernel_size=2, stride=2),          #  (256, 8, 8),  (256, 4, 4)
            nn.Dropout(0.25)                                #  (256, 4, 4),  (256, 4, 4)
        )

        #  
        self.classifier = nn.Sequential(
            nn.Linear(256 * 4 * 4, 512),                    #  (256*4*4),  (512)
            nn.ReLU(inplace=True),                          #  (512),  (512)
            nn.Dropout(0.5),                                #  (512),  (512)
            nn.Linear(512, 256),                            #  (512),  (256)
            nn.ReLU(inplace=True),                          #  (256),  (256)
            nn.Dropout(0.5),                                #  (256),  (256)
            nn.Linear(256, num_classes)                     #  (256),  (num_classes)
        )

    def forward(self, x):
        x = self.conv1(x)
        x = self.conv2(x)
        x = self.conv3(x)
        x = x.view(x.size(0), -1)
        x = self.classifier(x)
        return x

def train_model(model, trainloader, criterion, optimizer, epochs=50, verbose=True):
    history = {
        'train_loss': [],
        'train_accuracy': [],
        'epoch_time': []
    }

    for epoch in range(epochs):
        start_time = time.time()
        model.train()
        running_loss = 0.0
        correct = 0
        total = 0

        for i, (inputs, labels) in enumerate(trainloader):
            inputs, labels = inputs.to(device), labels.to(device)

            optimizer.zero_grad()

            outputs = model(inputs)
            loss = criterion(outputs, labels)
            loss.backward()
            optimizer.step()

            running_loss += loss.item()
            _, predicted = outputs.max(1)
            total += labels.size(0)
            correct += predicted.eq(labels).sum().item()

        scheduler.step()

        epoch_loss = running_loss / len(trainloader)
        epoch_acc = 100. * correct / total
        epoch_time = time.time() - start_time

        history['train_loss'].append(epoch_loss)
        history['train_accuracy'].append(epoch_acc)
        history['epoch_time'].append(epoch_time)

        if verbose:
            print(f" {epoch+1:02d}: "
                  f"Train loss={epoch_loss:.4f}, acc={epoch_acc:.2f}% | "
                  f": {epoch_time:.2f} c")

    return history

def evaluate_model(model, testloader):

    test_loss = 0
    correct = 0
    total = 0

    with torch.no_grad():
        for inputs, labels in testloader:
            inputs, labels = inputs.to(device), labels.to(device)
            outputs = model(inputs)
            loss = criterion(outputs, labels)

            test_loss += loss.item()
            _, predicted = outputs.max(1)
            total += labels.size(0)

    test_loss /= len(testloader)
    test_accuracy = 100. * correct / total

    print(f' : {test_loss:.4f}')

    return test_loss, test_accuracy

def show_predictions(model, testloader):
    model.eval()
    images, labels = next(iter(testloader))
    
    with torch.no_grad():
        predicted = model(images.to(device)).argmax(1).cpu()

    plt.figure(figsize=(12, 8))
    
    for i in range(12):
        plt.subplot(3, 4, i + 1)

        #

        img = img.numpy().transpose(1, 2, 0)    #  (C, H, W)  (H, W, C)
        img = np.clip(img, 0, 1)
        
        plt.imshow(img)
        
        true, pred = classes[labels[i]], classes[predicted[i]]
        plt.title(f'True {true}\nPred {pred}', color='green' if true == pred else 'red', fontsize=9)
        plt.axis('off')
    
    plt.tight_layout()
    plt.show()
